```javascript
// k6 load test script
import ws from 'k6/ws';
import { check } from 'k6';

export const options = {
  stages: [
    { duration: '2m', target: 10 }, // Ramp up
    { duration: '5m', target: 50 }, // Stay at 50 users
    { duration: '2m', target: 100 }, // Spike test
    { duration: '2m', target: 0 }, // Ramp down
  ],
};

export default function () {
  const url = 'wss://ws.wibblywobblas.com';
  const params = { tags: { name: 'DJ_Deck_Load_Test' } };
  
  const res = ws.connect(url, params, function (socket) {
    socket.on('open', () => {
      socket.send(JSON.stringify({
        type: 'join_session',
        session_id: 'test_session'
      }));
      
      // Simulate queue submission
      socket.send(JSON.stringify({
        type: 'submit_track',
        track: generateRandomTrack()
      }));
    });
    
    socket.on('message', (data) => {
      check(data, {
        'Message received': (d) => d !== null,
        'Valid JSON': (d) => JSON.parse(d) !== null
      });
    });
    
    socket.setTimeout(() => {
      socket.close();
    }, 30000); // 30 second sessions
  });
  
  check(res, { 'Connected successfully': (r) => r && r.status === 101 });
}
```

### Error Boundary Implementation
```typescript
class DJDeckErrorBoundary extends React.Component<Props, State> {
  state = { hasError: false, error: null };
  
  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }
  
  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    // Log to Sentry
    Sentry.captureException(error, {
      contexts: {
        react: { componentStack: errorInfo.componentStack }
      }
    });
    
    // Log to analytics
    trackEvent('dj_deck_error', {
      error: error.message,
      stack: error.stack
    });
  }
  
  render() {
    if (this.state.hasError) {
      return (
        <div className="error-fallback">
          <h2>DJ Deck temporarily unavailable</h2>
          <p>The main site is still functional.</p>
          <button onClick={() => window.location.href = '/'}>
            Return to Home
          </button>
        </div>
      );
    }
    
    return this.props.children;
  }
}
```

### Rollback Plan
```bash
#!/bin/bash
# rollback.sh

# 1. Disable feature flag immediately
curl -X POST $API_URL/feature-flags \
  -H "Authorization: Bearer $TOKEN" \
  -d '{"flag": "DJ_DECK_ENABLED", "value": false}'

# 2. Revert to previous deployment
vercel rollback $PREVIOUS_DEPLOYMENT_ID

# 3. Clear CDN cache
curl -X POST https://api.cloudflare.com/client/v4/zones/$ZONE_ID/purge_cache \
  -H "Authorization: Bearer $CF_TOKEN" \
  -d '{"purge_everything": true}'

# 4. Notify monitoring
curl -X POST $SLACK_WEBHOOK \
  -d '{"text": "DJ Deck feature rolled back"}'

# 5. Verify rollback
./scripts/health-check.sh
```
